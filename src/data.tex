\def\Json#1{\cs{invoice/#1}}
\def\JsonOrEmpty#1{\trycs{invoice/#1}{}}

\def\VatText#1{%
  \edef\tmp{#1}%
  \def\trueval{true}%
  \ifx\tmp\trueval Plátce DPH\else Neplátce DPH\fi
}

    \InvoiceRow
